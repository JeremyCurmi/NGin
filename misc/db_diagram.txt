//// -- LEVEL 1
//// -- Tables and References

// Creating tables
Table users as U {
  id int [pk, increment] // auto-increment
  name varchar
  created_at timestamp
}

Table models as M {
    id int [pk, increment] // auto-increment
    model_id int
    name varchar
    description varchar
    created_at timestamp
}

Table userModel as UM {
  user_id int
  model_id int
 }

Table modelTypes as MT {
  id int [pk, increment]
  name varchar
}

Table modelModelType as MMT {
  model_id int
  model_type_id int
 }

// > many-to-one; < one-to-many; - one-to-one
Ref: U.id < M.id
Ref: M.id > MT.id


Table modelVersions as MV {
  id int [pk, increment]
  // model_id int [ref: < M.model_id]
  model_version_id int
  description varchar
  created_at timestamp
}

Table modelModelVersion as MMV {
  model_id int
  model_version_id int
}

Ref: M.id < MV.id
